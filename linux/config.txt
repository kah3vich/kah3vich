//| Terminal: 

zsh - unix shells.
    theme -> oh my zsh ( powerlevel10k ) .zshrc

//| Command: 

date - return current date.

cal - return current calendar.

df - disk memory statistics.
    -a (--all) - return all files in system.
    -h - return list info in formatted megabytes or gigabytes.

free - free memory statistics.

exit - end session in terminal.

clear - clear current terminal.

pwd - return current path directory.

whatis <command> - return short info command.

file <file> - return type file.

type <command> - return type command.

help <command> - return full info it command.

which <file | command> - return info path from file or command.

alias <name alias>='<flow commands>' - create alias for system.

unalias <name alias> - remove alias in system.

cat <file> - read file and return data in terminal.
    -E - show the $ character at the end of each line.
    -n - number all lines.
    -s - delete empty duplicate lines.

less <file> - show content current file. ( keyboard q - exit )
    -E (--QUIT-AT-EOF) - exit if will reach end file.

sort <file> - sorted line string.
    -b - ignore spaces.
    -r - sort in reverse order.
    -c - check if file is sorted.
    -o - return result command.

uniq <file> - return message if have file empty line and remove empty line.
    -u - displays only those lines that do not have repetitions.
    -D - displays only duplicate lines.

cp <prev> <next> - copy file or directory.
    -l - create hard link.
    -i - overwrite request. 
    -n - do not overwrite existing files.
    -u - copy file if it has been modified.
    -b - create backup for old file.
    -v - return full info from result command.

mv <old> <new> - moving or rename -> file or directory.
    -b - create backup for old file.
    -f - don t ask permission for overwrite.
    -i - ask permission for overwrite.
    -u - performed if file it has been modified.
    -v - return full info from result command.

mkdir <directory name> - create new directory.
    -v - return info new directory.

rm <file | directory> - remove file or directory.
    -r - remove directory if there is something inside.
    -f - ignore if don t file or directory.
    -i - ask permission to remove.
    -d - remove directory if directory empty.
    -v - return full info from result command.

ln <file> - create hard link.
    <file | directory> -s - create symbolic link.
    -f - overwrite current file.

ls - return data list in current directory.
    <path> - return data list in given path. ( ./bin )
    -a (--all) - all data list ( secret file ).
    -d (--directory) - return info current directory. (-l)
    -l - return full info data.
    -h (--human-readable) - return (-l) info file size is not in bytes, but in units of measure.
    -r (--reverse) - return information in reverse order.
    -S - return sorted information by size.
    -t - return sorted list information by last modified.
    -X - return sorted list by alphabet.

cd <path> - switch in given path.
    ".." | "../" | "-" - back one level.
    "~" | empty - switch to system main directory.
    -e - return error message if path empty.

echo <content> - output content.
    $(($((5 + 2)) * 3)) - algorithmic calculations ( + | - | * | : | % | ** ).

//| Additional function: 

<command> 
    --help - return full info current command.
    --version - return the installed command version.

> - pass data from command to file.
    > <file> - create empty file.

| - transferring data from another command to the current.
