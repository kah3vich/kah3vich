//! Schema: 

git add . - indexing changes.

git commit -m "feat: init" - commit the change.

git push - sending fixed data (snapshots).


//! Command: 

git archive - create archive file from path.
    --format= - format for archive: zip | tar.
    -o (--output) - path file for archive.
    --prefix= - prefix for archive.

git init <directory> - initialization project.
     --initio-branch <name branch> - init repository with current branch.

git stash - save changes locally.
     save "name" - create a stesh with the name.
         -u - count all files and folders.
     apply stash@{number} - get stash with specified id.
     drop stash@{number} - remove stash with specified id.
     list - list of all stashes.
     clear - clear the entire list of stashes.

git add - indexing changes.

git commit - committing a change.
      -m - add a comment.
      -a - index changes (without git add ).
      --amend - cancel the change.

git reset HEAD [file] - cancel file indexing.

git branch - return list branch.
     -a - show all branch in repository.
     -d [name branch] - delete branch.
     [name branch] - create branch.
     -v - return list branch & last commit by branch.
     --no-merge - return list branch no merge other branch.
     --merged - merge branch with current branch.

git checkout - switch to a branch.
      -- [file] - undo changes made to the file.
      -b [name branch] - create branch.

git remote - remote repository.
      -v - list of remote repositories.
      add [name] [url] - adding a remote repository.
      show [name] - show repository server.
      rename [old] [new] - rename remote repository.

git tag - show list tag in repository.
     -a [name] - create tag.
     -m [message] - tag message.
     [number] - get current tag. (v1.0.0)
     -a [name] [commit] - create tag for commit.

git show [tag] - return short info tag.

git fetch [name] - fetch data from a remote repository.

git push (name server - origin) [branch] - sending data to a remote repository.
     --tags - sending all tags.
     -f - ignore git by error merge.
     -u - setting "upstream" or "merge candidate".

git request-pull [start commit] [end commit] [url repository] - used to send a merge request for changes from a local branch to a remote repository.

git diff - output line-by-line changes in files.
     --staged - display files in indexing.
     --check - check code by error space.

git status - displays the current status of files, indexing and committing.
     -s - abbreviated output.

git log - display information about versions.
     -p - show changes made to each version.
     -[number] - show the number of recent commits.
     --stat - show statistics of file changes in each commit.

git rm - delete a file for git.

git clone [url] [path (.)] - Clone a repository.
     --bare - the command is used to create a "bare" repository.
     -b [name branch] - clone repository with current branch.

git mv [path] [path] - move and rename files or folders.

git help - display information about git.

git merge [branch] - merge branch. (rebase)

git config
     --global - global config setting.
     --local - local config setting (in the current repository).
     --list - display general config information.
     user.name - specify the name for the config.
     user.email - specifying mail for the config.
     core.editor - set the main code editor ( vscode - git config --global core.editor "code --wait" | git config --global core.editor "code -w").
     alias.[name] 'code' - create alias for config git.
