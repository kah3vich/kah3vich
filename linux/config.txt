#!/bin/bash - specifying the interpreter that will execute the script.

//! function:

name_function() {
     return
}

function name_function() {
     return
}

name_function // function call

//! Variables:

name_variable=value // global variable

local name_variable=value // local variable

echo "foo -> $foo" // output variable

//! Условия: 

x=5 

if [$x = 5]; then 
    echo "x -> 5"
elif [$x = 1]; then 
    echo "x -> 1"
else 
    echo "x -> $x"
fi 

//! Loop:

count=1

while [[ $count -le 5 ]]; do // while true; do
     echo $count
     count=$((count + 1))
done

until - very similar to while, but terminates the loop not when it encounters a non-zero exit code until it receives an exit code of 0.

count=1

until [[ $count -gt 5 ]]; do
     echo $count
     count=$(( count + 1 ))
done

//! "For" loop:

for i in A B C D; do echo $i; done 

for i in {A..B}; do echo $i; done 

for i in distros*.txt; do echo $i; done 

for (( i=0; i<5; i=i+1 )); do echo $i; done 

//! Cases: 

 read -p "... > "

 case "$REPLY" in 
    [[:alpha:]]) echo "1" ;;
    [ABC][0-9]) echo "1" ;;
    a|A) echo "1" ;;
    ???) echo "1" ;;
    *.txt) echo "1" ;;
    *) echo "1" ;;
esac 

;; - completion of the check.
;;& - check other conditions.
