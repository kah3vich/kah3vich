//| Command: 

date - return current date.

cal - return current calendar.

df - disk memory statistics.
    -a (--all) - return all files in system.
    -h - return list info in formatted megabytes or gigabytes.

free - free memory statistics.

exit - end session in terminal.

clear - clear current terminal.

pwd - return current path directory.

whatis <command> - return short info command.

file <file> - return type file.

type <command> - return type command.

help <command> - return full info it command.

which <file | command> - return info path from file or command.

alias <name alias>='<flow commands>' - create alias for system.

unalias <name alias> - remove alias in system.

cat <file> - read file and return data in terminal.
    -E - show the $ character at the end of each line.
    -n - number all lines.
    -s - delete empty duplicate lines.
    -A - return non-printable characters in text.

less <file> - show content current file. ( keyboard q - exit )
    -E (--QUIT-AT-EOF) - exit if will reach end file.

sort <file> - sorted line string.
    -b - ignore spaces.
    -f - ignore case.
    -n - sort based on line numbering.
    --key=num - sort by key.
    -r - sort in reverse order.
    -c - check if file is sorted.
    -o - return result command.

uniq <file> - return message if have file empty line and remove empty line.
    -u - displays only those lines that do not have repetitions.
    -d - displays only duplicate lines.
    -c - return number duplicate in text.

cp <prev> <next> - copy file or directory.
    -l - create hard link.
    -i - overwrite request. 
    -n - do not overwrite existing files.
    -u - copy file if it has been modified.
    -b - create backup for old file.
    -v - return full info from result command.

mv <old> <new> - moving or rename -> file or directory.
    -b - create backup for old file.
    -f - don t ask permission for overwrite.
    -i - ask permission for overwrite.
    -u - performed if file it has been modified.
    -v - return full info from result command.

mkdir <directory name> - create new directory.
    -v - return info new directory.

rm <file | directory> - remove file or directory.
    -r - remove directory if there is something inside.
    -f - ignore if don t file or directory.
    -i - ask permission to remove.
    -d - remove directory if directory empty.
    -v - return full info from result command.

ln <file> - create hard link.
    <file | directory> -s - create symbolic link.
    -f - overwrite current file.

ls - return data list in current directory.
    <path> - return data list in given path. ( ./bin )
    -a (--all) - all data list ( secret file ).
    -d (--directory) - return info current directory. (-l)
    -l - return full info data.
    -h (--human-readable) - return (-l) info file size is not in bytes, but in units of measure.
    -r (--reverse) - return information in reverse order.
    -S - return sorted information by size.
    -t - return sorted list information by last modified.
    -X - return sorted list by alphabet.

cd <path> - switch in given path.
    ".." | "../" | "-" - back one level.
    "~" | empty - switch to system main directory.
    -e - return error message if path empty.

wc <file> - file data analysis. Displays a count of the sizes and volumes of files.
    -c - return size file in bit.
    -m return count symbol in object.
    -l - return count lines in object.
    -w - return count word in object.

uniq <file> - search for identical lines in a text array.
    -u - outputs lines that do not have repetitions.
    -d - output repeated lines once.
    -D - output only duplicate lines.
    -i - ignore register.
    -c - print the number of repetitions.
    -z - split for output.

grep <string> <file> - search in a file.
    -f - take pattern from file.
    -i - case insensitive.
    -w - displays everything except found lines.
    -c - displays the number of lines found.
    -m - stop searching after reaching the specified number.
    -s - don't output read error.
    -h - don't print filename.

    [:alnum:] - alphanumeric characters [A-Za-z0-9].
    [:word:] - like [:alnum:], but with an additional character (_).
    [:alpha:] - alphabetic characters [A-Za-z].
    [:blank:] - includes space and tab characters.
    [:digit:] - digits from 0 to 9.
    [:lower:] - lower case characters.
    [:punct:] - punctuation marks.
    [:print:] - all characters from the class [:graph:] and a space.
    [:space:] - characters [ \t \r \n \v \f ].
    [:upper:] - upper case characters.
    [:xdigit:] - characters for hexadecimal digits [0-9A-Fa-f].

clear - clear the terminal.

head <file> - return first string in file.
    -q - return only text.
    -v - print file names before text.
    -n <count> - output a limited number of lines.
    -f - streaming.

tail <file> - return last string in file.
    -q - return only text.
    -v - display additional information about the file.
    -n <count> - output a limited number of lines.
    -f - streaming.

tee <file> - saves the output of a command to a file (via a pipeline).
    -a - write data to the end of the file.

history - command history output ( max-base 500 or 1000 ).
    -d <number> - delete a specific command.
    -c - clear history.
    -w <file> - write commands to a file.
    !<number> - call a command under a specific number.
    !! - call the previous command.
    <number> - limited list from the beginning (last commands can be received from a negative number).

id - returns actual data of system users.
     -u - returns the actual user indicator.
     -g - returns the actual indicator of the user's group.

chmod <options> <permissions> <file> - command to control file permissions.
     r - read.
     w - entry.
     x - execution.
     s - run as superuser (optional).
     u is the owner of the file.
     g - file group.
     o - all other users.

     u+x - allow execution for the owner.
     ugo+x - allow execution for everyone.
     ug+w - enable write for owner and group.
     o-x - disable execution for other users.
     ugo+rwx - allow all for everyone.

umask <options> <permissions> - used to set default permissions. (umask u=rwx)
     u (user) – user.
     g (group) – group.
     o (other) – other users.

     r (read) - read file.
     w (write) - change the file.
     x (execute) - execute the file as a program.

su - Changes the shell user to the specified one.
     [-] - change the shell execution context to the context of the specified user.
     -c - runs the application under the specified account.
    
sudo - runs the command passed to it as superuser.
     -b Runs the given program in the background.

chown <option> <user> <file> - changes access to a file or directories in the system.
     -c Verbose output of all changes being made.
     -f - minimum information, hide error messages.
     -H - if a symbolic link to a directory is passed, navigate to it.
     -L - follow all symbolic links to directories.

passwd <user> - Changes the user's password.
     -d - Delete the user's password (make it blank). This is a quick way to lock out an account password.
     -e Make the password obsolete immediately. As a result, this will force the user to change the password the next time they log in.
     -w <number> - Set the number of warning days before the password expires.

ps - returns up-to-date data about system processes.
     -A - select all processes.
     -a - select all processes except background ones.
     -d - select all processes, even background ones, except session processes.
     -C <name> - select processes by command name.

top - Returns real-time system data.
     -v - display the version of the program.
     -c - display full path to executable command files.
     -d - information update interval.
     -i Do not display processes that are not using CPU resources.

jobs - displays a list of processes that are running in the background.

fg %<number> - brings the process to the foreground.

bg %<number> - sends the process to the background.

kill <signal> <process> - command to send signals to processes for state (terminating a process).
     -l - get a list of signals supported by the system.
     -s - use signal name.
     -n - use signal number.

killall <process> - намеренное завершение процесса.

shutdown - завершить работу компьютера.
    -h <now | +number | 21:00> - через сколько выключить.
    -c - отмена выключения.

neofetch - выводит краткую информацию о текущем дистрибутиве.

halt - выключение питания компьютера.

poweroff - аналог "halt".

reboot - перезагрузка компьютера. ( используется с "sudo" )

pstree - показывает дерево процессов.

vmstat - live report virtual memory statistics.
    -t - добавление временной метки.

echo <content> - output content.
    -n - don't output newline.
    -e - enable Escape.
    -E - disable Escape.

    /c - remove newline.
    /t - tabulation.
    /v - vertical tab.
    /b - delete previous character.
    /n - line break.
    /r - carriage return at the beginning of the line.
    \a - sound notification.
    \d - current date.
    \h - current machine.
    \H - fully qualified hostname.
    \t - current time.
    \u - username.
    \w - current directory.
    \! - team number in history.
    \# - number of commands in the current session.

    * - any character.
    [[:upper:]] - upper case.
    ~ - system directive.
    $USER - variable.
    $(($((5 + 2)) * 3)) - algorithmic calculations ( + | - | * | : | % | ** ).
    \$ - character escape.
    \\ - escaping a stash.

<command> 
    --help - return full info current command.
    --version - return the installed command version.
    & - running the program in the background.

    > <file> - create empty file.

>> redirect to end of file

2> redirect (even errors) to a file

&> double redirect with error handling to file

&>> ...end of file

< redirecting data from a file ( cat < file.txt )
